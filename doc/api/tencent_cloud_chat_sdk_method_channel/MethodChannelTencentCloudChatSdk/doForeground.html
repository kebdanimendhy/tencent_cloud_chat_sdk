@override
Future<V2TimCallback> doForeground() async {
  Map<String, dynamic> param = {
    "ability": Utils.getAbility(),
  };
  var resp = V2TimCallback.fromJson(
    formatJson(
      await _channel.invokeMethod(
        "doForeground",
        param,
      ),
    ),
  );
  log("doForeground", param, resp.toLogString());
  return resp;
}
