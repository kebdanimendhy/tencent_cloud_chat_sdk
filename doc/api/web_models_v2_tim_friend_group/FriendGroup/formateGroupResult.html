static List<dynamic> formateGroupResult(list) {
  List<dynamic> resultArr = [];
  list.forEach((item) {
    final formatedItem = jsToMap(item);
    resultArr.add({
      "name": formatedItem['name'],
      "friendCount": formatedItem['count'],
      "friendIDList": formatedItem['userIDList'],
    });
  });

  return resultArr;
}
